# Quick Database Export Commands for Server
# Run these commands on your Ubuntu server (51.21.181.162)

# 1. Connect to your server
# ssh -i TucanBIT.pem ubuntu@51.21.181.162

# 2. Create backup directory
# mkdir -p /opt/tucanbit/backups

# 3. Export database (run these commands on the server)
# export PGPASSWORD='5kj0YmV5FKKpU9D50B7yH5A'
# pg_dump -h localhost -p 5432 -U tucanbit -d tucanbit --no-owner --no-privileges --file=/opt/tucanbit/backups/tucanbit_export_$(date +%Y%m%d_%H%M%S).sql

# 4. Export in custom format (recommended)
# pg_dump -h localhost -p 5432 -U tucanbit -d tucanbit --format=custom --no-owner --no-privileges --file=/opt/tucanbit/backups/tucanbit_export_$(date +%Y%m%d_%H%M%S).dump

# 5. Download to your local machine
# scp -i TucanBIT.pem ubuntu@51.21.181.162:/opt/tucanbit/backups/tucanbit_export_*.sql .
# scp -i TucanBIT.pem ubuntu@51.21.181.162:/opt/tucanbit/backups/tucanbit_export_*.dump .
